IF [Value]<0 THEN '-' ELSE '' END
+
'$'
+
STR(INT(ROUND(ABS([Value])/POWER(1000,INT(LOG(ROUND(ABS([Value])/10,0)*10,1000))),1)))
+
IF (ROUND(ABS([Value])/POWER(1000,INT(LOG(ABS([Value]),1000))),1)*10)%10=0
THEN ''
ELSE '.' + STR((ROUND(ABS([Value])/POWER(1000,INT(LOG(ABS([Value]),1000))),1)*10)%10)
END
+
CASE 
INT(LOG(ROUND(ABS([Value])/10,0)*10,1000))
WHEN 1 THEN 'K'
WHEN 2 THEN 'M'
WHEN 3 THEN 'B'
WHEN 4 THEN 'T'
ELSE ''
END
